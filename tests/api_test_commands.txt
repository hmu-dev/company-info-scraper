
# Test the API with curl

# 1. Health check:
curl -X GET http://localhost:8000/health

# 2. Scrape profile only:
curl -X POST http://localhost:8000/scrape/profile \
  -H "Content-Type: application/json" \
  -d "{
    \"url\": \"https://www.openai.com\",
    \"model\": \"gpt-3.5-turbo\"
  }"

# 3. Scrape media only:
curl -X POST http://localhost:8000/scrape/media \
  -H "Content-Type: application/json" \
  -d "{
    \"url\": \"https://www.openai.com\",
    \"model\": \"gpt-3.5-turbo\"
  }"

# 4. Scrape combined (profile + media):
curl -X POST http://localhost:8000/scrape/combined \
  -H "Content-Type: application/json" \
  -d "{
    \"url\": \"https://www.openai.com\",
    \"model\": \"gpt-3.5-turbo\"
  }"
